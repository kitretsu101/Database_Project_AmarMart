

DROP DATABASE IF EXISTS swiftmart_db;
CREATE DATABASE swiftmart_db
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;

USE swiftmart_db;


CREATE TABLE roles (
    id          INT AUTO_INCREMENT PRIMARY KEY,
    name        VARCHAR(50)  NOT NULL UNIQUE,
    slug        VARCHAR(50)  NOT NULL UNIQUE,
    permissions JSON         DEFAULT NULL COMMENT 'JSON array of permission strings',
    created_at  TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    updated_at  TIMESTAMP    DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB;


CREATE TABLE branches (
    id          INT AUTO_INCREMENT PRIMARY KEY,
    name        VARCHAR(100) NOT NULL,
    address     VARCHAR(255) DEFAULT NULL,
    phone       VARCHAR(30)  DEFAULT NULL,
    email       VARCHAR(100) DEFAULT NULL,
    emoji       VARCHAR(10)  DEFAULT '🏪',
    status      ENUM('Active','Inactive') DEFAULT 'Active',
    created_at  TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    updated_at  TIMESTAMP    DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_status (status)
) ENGINE=InnoDB;

CREATE TABLE users (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    full_name       VARCHAR(100) NOT NULL,
    email           VARCHAR(100) NOT NULL UNIQUE,
    password_hash   VARCHAR(255) NOT NULL,
    phone           VARCHAR(30)  DEFAULT NULL,
    role_id         INT          NOT NULL,
    branch_id       INT          DEFAULT NULL,
    salary          DECIMAL(12,2) DEFAULT 0.00,
    shift           ENUM('Morning','Evening','Night') DEFAULT 'Morning',
    avatar_color    VARCHAR(100) DEFAULT 'linear-gradient(135deg,#2563EB,#1D4ED8)',
    attendance_pct  DECIMAL(5,2) DEFAULT 0.00,
    is_active       TINYINT(1)   DEFAULT 1,
    remember_token  VARCHAR(255) DEFAULT NULL,
    last_login      TIMESTAMP    NULL,
    created_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (role_id)   REFERENCES roles(id),
    FOREIGN KEY (branch_id) REFERENCES branches(id) ON DELETE SET NULL,
    INDEX idx_email    (email),
    INDEX idx_role     (role_id),
    INDEX idx_branch   (branch_id),
    INDEX idx_active   (is_active)
) ENGINE=InnoDB;

-- Add manager_id to branches (circular FK, added after users table)
ALTER TABLE branches ADD COLUMN manager_id INT DEFAULT NULL;
ALTER TABLE branches ADD FOREIGN KEY (manager_id) REFERENCES users(id) ON DELETE SET NULL;


CREATE TABLE categories (
    id          INT AUTO_INCREMENT PRIMARY KEY,
    name        VARCHAR(100) NOT NULL UNIQUE,
    icon        VARCHAR(10)  DEFAULT '📦',
    description TEXT         DEFAULT NULL,
    is_active   TINYINT(1)   DEFAULT 1,
    created_at  TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    updated_at  TIMESTAMP    DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB;


CREATE TABLE products (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    name            VARCHAR(150) NOT NULL,
    barcode         VARCHAR(50)  DEFAULT NULL UNIQUE,
    sku             VARCHAR(50)  DEFAULT NULL UNIQUE,
    category_id     INT          NOT NULL,
    brand           VARCHAR(100) DEFAULT NULL,
    purchase_price  DECIMAL(12,2) DEFAULT 0.00,
    selling_price   DECIMAL(12,2) NOT NULL,
    unit            VARCHAR(20)  DEFAULT 'pcs',
    image           VARCHAR(255) DEFAULT NULL,
    emoji           VARCHAR(10)  DEFAULT '📦',
    description     TEXT         DEFAULT NULL,
    vat_rate        DECIMAL(5,2) DEFAULT 5.00,
    discount        DECIMAL(5,2) DEFAULT 0.00,
    location        VARCHAR(100) DEFAULT NULL COMMENT 'Aisle/Shelf location',
    expiry_date     DATE         DEFAULT NULL,
    is_active       TINYINT(1)   DEFAULT 1,
    created_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (category_id) REFERENCES categories(id),
    INDEX idx_category (category_id),
    INDEX idx_barcode  (barcode),
    INDEX idx_name     (name),
    INDEX idx_active   (is_active)
) ENGINE=InnoDB;

CREATE TABLE branch_products (
    id          INT AUTO_INCREMENT PRIMARY KEY,
    branch_id   INT NOT NULL,
    product_id  INT NOT NULL,
    quantity    INT DEFAULT 0,
    min_stock   INT DEFAULT 10,
    max_stock   INT DEFAULT 1000,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (branch_id)  REFERENCES branches(id) ON DELETE CASCADE,
    FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE CASCADE,
    UNIQUE KEY uk_branch_product (branch_id, product_id),
    INDEX idx_quantity (quantity)
) ENGINE=InnoDB;

CREATE TABLE customers (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    full_name       VARCHAR(100) NOT NULL,
    phone           VARCHAR(30)  DEFAULT NULL,
    email           VARCHAR(100) DEFAULT NULL,
    address         TEXT         DEFAULT NULL,
    dob             DATE         DEFAULT NULL,
    loyalty_points  INT          DEFAULT 0,
    tier            ENUM('Bronze','Silver','Gold','Platinum') DEFAULT 'Bronze',
    total_spent     DECIMAL(14,2) DEFAULT 0.00,
    total_orders    INT          DEFAULT 0,
    is_active       TINYINT(1)   DEFAULT 1,
    created_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_phone (phone),
    INDEX idx_name  (full_name),
    INDEX idx_tier  (tier)
) ENGINE=InnoDB;


CREATE TABLE sales (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    invoice_no      VARCHAR(30)  NOT NULL UNIQUE,
    customer_id     INT          DEFAULT NULL,
    user_id         INT          NOT NULL COMMENT 'Cashier',
    branch_id       INT          NOT NULL,
    subtotal        DECIMAL(14,2) DEFAULT 0.00,
    discount_pct    DECIMAL(5,2)  DEFAULT 0.00,
    discount_amount DECIMAL(14,2) DEFAULT 0.00,
    vat_amount      DECIMAL(14,2) DEFAULT 0.00,
    grand_total     DECIMAL(14,2) DEFAULT 0.00,
    payment_method  ENUM('Cash','Card','bKash','Nagad') DEFAULT 'Cash',
    status          ENUM('Paid','Pending','Cancelled','Refunded') DEFAULT 'Paid',
    notes           TEXT         DEFAULT NULL,
    created_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (customer_id) REFERENCES customers(id) ON DELETE SET NULL,
    FOREIGN KEY (user_id)     REFERENCES users(id),
    FOREIGN KEY (branch_id)   REFERENCES branches(id),
    INDEX idx_invoice   (invoice_no),
    INDEX idx_customer  (customer_id),
    INDEX idx_user      (user_id),
    INDEX idx_branch    (branch_id),
    INDEX idx_date      (created_at),
    INDEX idx_status    (status)
) ENGINE=InnoDB;


CREATE TABLE sale_items (
    id          INT AUTO_INCREMENT PRIMARY KEY,
    sale_id     INT NOT NULL,
    product_id  INT NOT NULL,
    quantity    INT NOT NULL DEFAULT 1,
    unit_price  DECIMAL(12,2) NOT NULL,
    total_price DECIMAL(14,2) NOT NULL,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (sale_id)    REFERENCES sales(id) ON DELETE CASCADE,
    FOREIGN KEY (product_id) REFERENCES products(id),
    INDEX idx_sale    (sale_id),
    INDEX idx_product (product_id)
) ENGINE=InnoDB;

CREATE TABLE payments (
    id          INT AUTO_INCREMENT PRIMARY KEY,
    sale_id     INT NOT NULL,
    method      ENUM('Cash','Card','bKash','Nagad') DEFAULT 'Cash',
    amount      DECIMAL(14,2) NOT NULL,
    reference   VARCHAR(100) DEFAULT NULL,
    status      ENUM('Completed','Pending','Failed') DEFAULT 'Completed',
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (sale_id) REFERENCES sales(id) ON DELETE CASCADE,
    INDEX idx_sale (sale_id)
) ENGINE=InnoDB;


CREATE TABLE suppliers (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    company_name    VARCHAR(150) NOT NULL,
    category        VARCHAR(100) DEFAULT NULL,
    contact_person  VARCHAR(100) DEFAULT NULL,
    phone           VARCHAR(30)  DEFAULT NULL,
    email           VARCHAR(100) DEFAULT NULL,
    city            VARCHAR(100) DEFAULT NULL,
    address         TEXT         DEFAULT NULL,
    emoji           VARCHAR(10)  DEFAULT '🏭',
    is_active       TINYINT(1)   DEFAULT 1,
    created_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_active (is_active)
) ENGINE=InnoDB;

CREATE TABLE purchase_orders (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    po_number       VARCHAR(30) NOT NULL UNIQUE,
    supplier_id     INT         NOT NULL,
    total_amount    DECIMAL(14,2) DEFAULT 0.00,
    order_date      DATE        NOT NULL,
    expected_date   DATE        DEFAULT NULL,
    received_date   DATE        DEFAULT NULL,
    status          ENUM('Pending','In Transit','Delivered','Cancelled') DEFAULT 'Pending',
    notes           TEXT        DEFAULT NULL,
    created_by      INT         DEFAULT NULL,
    created_at      TIMESTAMP   DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP   DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (supplier_id) REFERENCES suppliers(id),
    FOREIGN KEY (created_by)  REFERENCES users(id) ON DELETE SET NULL,
    INDEX idx_supplier (supplier_id),
    INDEX idx_status   (status),
    INDEX idx_date     (order_date)
) ENGINE=InnoDB;

CREATE TABLE po_items (
    id          INT AUTO_INCREMENT PRIMARY KEY,
    po_id       INT NOT NULL,
    product_id  INT DEFAULT NULL,
    description VARCHAR(200) DEFAULT NULL,
    quantity    INT NOT NULL DEFAULT 1,
    unit_price  DECIMAL(12,2) DEFAULT 0.00,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (po_id)      REFERENCES purchase_orders(id) ON DELETE CASCADE,
    FOREIGN KEY (product_id) REFERENCES products(id) ON DELETE SET NULL,
    INDEX idx_po (po_id)
) ENGINE=InnoDB;

CREATE TABLE notifications (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    user_id         INT          DEFAULT NULL COMMENT 'NULL = broadcast to all',
    type            VARCHAR(50)  NOT NULL DEFAULT 'general' COMMENT 'Stock, Expiry, Delivery, System, Sale, Product',
    title           VARCHAR(200) NOT NULL,
    message         TEXT         DEFAULT NULL,
    severity        ENUM('info','success','warning','danger') DEFAULT 'info',
    is_read         TINYINT(1)   DEFAULT 0,
    related_id      INT          DEFAULT NULL,
    related_type    VARCHAR(50)  DEFAULT NULL COMMENT 'sale, product, po, branch, etc.',
    created_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    INDEX idx_user   (user_id),
    INDEX idx_read   (is_read),
    INDEX idx_type   (type),
    INDEX idx_date   (created_at)
) ENGINE=InnoDB;

CREATE TABLE stock_movements (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    branch_product_id INT       NOT NULL,
    type            ENUM('in','out','transfer','damage','return') NOT NULL,
    quantity        INT         NOT NULL,
    reference       VARCHAR(100) DEFAULT NULL COMMENT 'PO number, Invoice number, etc.',
    notes           TEXT        DEFAULT NULL,
    user_id         INT         DEFAULT NULL,
    created_at      TIMESTAMP   DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (branch_product_id) REFERENCES branch_products(id) ON DELETE CASCADE,
    FOREIGN KEY (user_id)           REFERENCES users(id) ON DELETE SET NULL,
    INDEX idx_type (type),
    INDEX idx_date (created_at)
) ENGINE=InnoDB;


CREATE TABLE promotions (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    name            VARCHAR(150) NOT NULL,
    type            ENUM('Flash','Seasonal','Bundle','Clearance') DEFAULT 'Flash',
    discount_value  VARCHAR(50)  DEFAULT NULL COMMENT 'e.g. 30%, Buy 2 Get 1, ৳141 off',
    discount_pct    DECIMAL(5,2) DEFAULT 0.00,
    start_date      DATE         NOT NULL,
    end_date        DATE         NOT NULL,
    applicable_products TEXT     DEFAULT NULL COMMENT 'Comma-separated product IDs or category names',
    usage_count     INT          DEFAULT 0,
    is_active       TINYINT(1)   DEFAULT 1,
    gradient_color  VARCHAR(100) DEFAULT 'linear-gradient(135deg,#2563EB,#7C3AED)',
    emoji           VARCHAR(10)  DEFAULT '🎯',
    banner_text     TEXT         DEFAULT NULL,
    created_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_active (is_active),
    INDEX idx_dates  (start_date, end_date)
) ENGINE=InnoDB;


CREATE TABLE settings (
    id          INT AUTO_INCREMENT PRIMARY KEY,
    config_key  VARCHAR(100) NOT NULL UNIQUE,
    config_value TEXT        DEFAULT NULL,
    category    VARCHAR(50)  DEFAULT 'general',
    created_at  TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    updated_at  TIMESTAMP    DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_key      (config_key),
    INDEX idx_category (category)
) ENGINE=InnoDB;

CREATE TABLE audit_logs (
    id          INT AUTO_INCREMENT PRIMARY KEY,
    user_id     INT          DEFAULT NULL,
    action      VARCHAR(100) NOT NULL,
    target_type VARCHAR(50)  DEFAULT NULL,
    target_id   INT          DEFAULT NULL,
    details     TEXT         DEFAULT NULL,
    ip_address  VARCHAR(45)  DEFAULT NULL,
    created_at  TIMESTAMP    DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL,
    INDEX idx_user   (user_id),
    INDEX idx_action (action),
    INDEX idx_date   (created_at)
) ENGINE=InnoDB;


CREATE TABLE attendance (
    id          INT AUTO_INCREMENT PRIMARY KEY,
    user_id     INT  NOT NULL,
    att_date    DATE NOT NULL,
    clock_in    TIME DEFAULT NULL,
    clock_out   TIME DEFAULT NULL,
    status      ENUM('Present','Absent','Leave','Half-Day') DEFAULT 'Present',
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    UNIQUE KEY uk_user_date (user_id, att_date),
    INDEX idx_date (att_date)
) ENGINE=InnoDB;


CREATE TABLE loyalty_rules (
    id          INT AUTO_INCREMENT PRIMARY KEY,
    min_spend   DECIMAL(12,2) DEFAULT 0.00,
    points      INT           DEFAULT 0,
    note        VARCHAR(150)  DEFAULT NULL,
    is_active   TINYINT(1)    DEFAULT 1,
    created_at  TIMESTAMP     DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB;


CREATE TABLE redemption_offers (
    id              INT AUTO_INCREMENT PRIMARY KEY,
    points_required INT          NOT NULL,
    reward          VARCHAR(200) NOT NULL,
    badge_class     VARCHAR(50)  DEFAULT 'badge-bronze',
    is_active       TINYINT(1)   DEFAULT 1,
    created_at      TIMESTAMP    DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB;
